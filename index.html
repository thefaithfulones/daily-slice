    const payload = {

📖 "${data.verse}" — ${data.verse}

💬 ${data.reflection}

🙏 ${data.prayer}

🕊️ May this slice nourish you too.`

    };

    await fetch('https://discord.com/api/webhooks/1379584389511647272/IsskAhyUK2rfkZP4RpF_YRw91Bobbvyy9HcmG02-hlm7T6rNqkMSajd3vNUIq4GAsBNL', {

      method: 'POST',

      headers: {'Content-Type': 'application/json'},

      body: JSON.stringify(payload)

    });

    alert('Your Daily Slice has been posted!');

    e.target.reset();

  });

</script>
